def debug(print_headers)
  puts "<-- Start Province debugging!" if print_headers
  print "Province #{old_id} => "
  @new_id.each { |new_prov| print "#{new_prov}, " }
  puts "controller: #{@controller}"
  puts "     End Province debugging!  -->" if print_headers
end
